% Generic section heading printer
\newcommand{\resumesectionheading}[3]{%
  % #1 = space before, #2 = heading text, #3 = space after
  \vspace{#1}%
  \noindent{\sectionfont #2}\par
  \vspace{#3}%
}

% Shared bullet-list environment (education, work, projects)
\newenvironment{resumeitems}{%
  \begin{itemize}[
    leftmargin=\ResumeBulletLeftMargin,
    labelsep=\ResumeBulletLabelSep,
    topsep=\ResumeBulletTopSep,
    partopsep=\ResumeBulletPartopsep,
    parsep=\ResumeBulletParsep,
    itemsep=\ResumeBulletItemSep,
    label=\raisebox{0.15ex}{\fontsize{6}{6}\selectfont$\bullet$}
  ]
}{%
  \end{itemize}%
  \vspace{\ResumeBulletBottomSep}%
}
